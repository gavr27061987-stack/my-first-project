    <script>
        // Простая логика для демонстрации
        document.querySelector('.search-btn').addEventListener('click', function() {
            const brand = document.getElementById('brand').value;
            const price = document.getElementById('price').value;
            
            if (!brand) {
                alert('Пожалуйста, выберите марку автомобиля');
                return;
            }
            
            let message = 'Поиск автомобилей ';
            message += brand ? марки ${brand} : '';
            message += price ? ` до $${price}` : '';
            
            alert(message);
        });
        
        // Обновление списка моделей при выборе марки
        document.getElementById('brand').addEventListener('change', function() {
            const brand = this.value;
            const modelSelect = document.getElementById('model');
            
            // Очищаем список моделей
            modelSelect.innerHTML = '<option value="">Любая модель</option>';
            
            if (!brand) return;
            
            // Добавляем модели в зависимости от выбранной марки
            let models = [];
            
            switch(brand) {
                case 'geely':
                    models = ['Coolray', 'Emgrand', 'Atlas', 'Tugella'];
                    break;
                case 'chery':
                    models = ['Tiggo 8 Pro', 'Arrizo 6', 'Tiggo 7 Pro', 'Tiggo 4'];
                    break;
                case 'haval':
                    models = ['Jolion', 'H6', 'F7', 'Big Dog'];
                    break;
                case 'byd':
                    models = ['Han', 'Tang', 'Song', 'Dolphin'];
                    break;
                case 'changan':
                    models = ['CS75 Plus', 'UNI-K', 'Eado', 'CS35 Plus'];
                    break;
            }
            
            models.forEach(model => {
                const option = document.createElement('option');
                option.value = model.toLowerCase().replace(' ', '-');
                option.textContent = model;
                modelSelect.appendChild(option);
            });
        });
    </script>
